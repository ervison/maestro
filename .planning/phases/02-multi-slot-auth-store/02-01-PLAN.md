```python
def test_load_backward_compat(auth_file):
    """load() returns TokenSet | None - backward compat for agent.py"""
    from maestro import auth
    from maestro.auth import TokenSet
    
    # No credentials stored
    assert auth.load() is None
    
    # Store credentials via new API
    auth.set("chatgpt", {
        "access": "tok",
        "refresh": "ref",
        "expires": 9999999.0,
        "account_id": "acc",
        "email": "test@example.com",
    })
    
    # load() should return TokenSet
    ts = auth.load()
    assert isinstance(ts, TokenSet)
    assert ts.access == "tok"
    assert ts.email == "test@example.com"


def test_save_backward_compat(auth_file):
    """_save(TokenSet) writes data readable by load()"""
    from maestro import auth
    from maestro.auth import TokenSet, _save
    
    ts = TokenSet(
        access="tok",
        refresh="ref",
        expires=9999999.0,
        account_id="acc",
        email="test@example.com",
    )
    _save(ts)
    
    # load() should read it back
    loaded = auth.load()
    assert loaded is not None
    assert loaded.access == "tok"
    assert loaded.email == "test@example.com"
```
