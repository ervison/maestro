```python
from importlib.metadata import entry_points
from functools import lru_cache

@lru_cache(maxsize=1)
def discover_providers() -> dict[str, ProviderPlugin]:
    """Discover all registered providers via entry points.
    
    Returns:
        Dict mapping provider_id to ProviderPlugin instance.
        
    Note: Entry points only resolve after pip install. If developing,
    run `pip install -e .` after modifying pyproject.toml entry points.
    """
    result: dict[str, ProviderPlugin] = {}
    for ep in entry_points(group="maestro.providers"):
        cls = ep.load()
        instance = cls()
        # Validate it satisfies ProviderPlugin protocol
        if isinstance(instance, ProviderPlugin):
            result[instance.id] = instance
    return result

def get_provider(provider_id: str) -> ProviderPlugin:
    """Get a provider by ID.
    
    Args:
        provider_id: The provider identifier (e.g., 'chatgpt', 'github-copilot')
        
    Returns:
        ProviderPlugin instance
        
    Raises:
        ValueError: If provider_id is not found. Message includes available IDs.
    """
    providers = discover_providers()
    if provider_id not in providers:
        available = sorted(providers.keys())
        raise ValueError(
            f"Unknown provider: {provider_id!r}. Available: {available}"
        )
    return providers[provider_id]
```

```python
"""Configuration and model resolution for maestro.

Model resolution order (highest priority first):
1. --model CLI flag (passed as argument)
2. MAESTRO_MODEL environment variable
3. config.agent.<agent_name>.model (when running a named agent)
4. config.model (global default)
5. First model of first authenticated provider (automatic fallback)

When no authenticated provider exists, falls back to ChatGPT default
to preserve backward compatibility with existing `maestro run` behavior.
"""

import json
import os
from pathlib import Path
from typing import Any

from maestro import auth
from maestro.providers import get_provider, discover_providers
from maestro.providers.chatgpt import DEFAULT_MODEL as CHATGPT_DEFAULT

CONFIG_FILE = Path(
    os.environ.get("MAESTRO_CONFIG_FILE", Path.home() / ".maestro" / "config.json")
)


def load_config() -> dict[str, Any]:
    """Load config from ~/.maestro/config.json.
    
    Returns:
        Config dict, or empty dict if file doesn't exist or is invalid.
    """
    if not CONFIG_FILE.exists():
        return {}
    try:
        return json.loads(CONFIG_FILE.read_text())
    except (json.JSONDecodeError, OSError):
        return {}


def parse_model(model: str) -> tuple[str, str]:
    """Parse 'provider_id/model_id' format.
    
    Args:
        model: Model string in 'provider/model' format
        
    Returns:
        Tuple of (provider_id, model_id)
        
    Raises:
        ValueError: If format is invalid (missing slash or empty parts)
    """
    provider_id, sep, model_id = model.partition("/")
    if not sep or not provider_id or not model_id:
        raise ValueError(
            f"Model must be in format 'provider/model', got: {model!r}. "
            f"Example: 'chatgpt/gpt-5.4-mini' or 'github-copilot/gpt-4o'"
        )
    return provider_id, model_id


def resolve_model(
    cli_model: str | None = None,
    agent_name: str | None = None,
) -> tuple[str, str]:
    """Resolve model using priority chain.
    
    Priority order:
    1. cli_model argument (from --model flag)
    2. MAESTRO_MODEL environment variable
    3. config.agent.<agent_name>.model (if agent_name provided)
    4. config.model (global default)
    5. First authenticated provider's first model
    6. ChatGPT default (backward compat fallback per D-02, D-03)
    
    Args:
        cli_model: Model from --model CLI flag (format: 'provider/model')
        agent_name: Optional agent name for per-agent config lookup
        
    Returns:
        Tuple of (provider_id, model_id)
        
    Raises:
        ValueError: If model format is invalid
    """
    # 1. CLI flag
    if cli_model:
        return parse_model(cli_model)
    
    # 2. Environment variable
    env_model = os.environ.get("MAESTRO_MODEL")
    if env_model:
        return parse_model(env_model)
    
    # 3 & 4. Config file
    config = load_config()
    
    # 3. Per-agent config
    if agent_name:
        agent_config = config.get("agent", {}).get(agent_name, {})
        agent_model = agent_config.get("model")
        if agent_model:
            return parse_model(agent_model)
    
    # 4. Global config default
    global_model = config.get("model")
    if global_model:
        return parse_model(global_model)
    
    # 5. First authenticated provider's first model
    providers = discover_providers()
    for provider in providers.values():
        if provider.is_authenticated():
            models = provider.list_models()
            if models:
                return provider.id, models[0]
    
    # 6. ChatGPT default fallback (per D-02, D-03 - preserve current behavior)
    return "chatgpt", CHATGPT_DEFAULT
```

```python
import pytest
from maestro.providers import (
    discover_providers,
    get_provider,
    ProviderPlugin,
)


class TestDiscoverProviders:
    def test_returns_dict(self):
        """discover_providers() returns a dict."""
        providers = discover_providers()
        assert isinstance(providers, dict)

    def test_chatgpt_discovered(self):
        """ChatGPT provider is discovered via entry points."""
        providers = discover_providers()
        assert "chatgpt" in providers

    def test_providers_are_instances(self):
        """Discovered providers are ProviderPlugin instances."""
        providers = discover_providers()
        for provider in providers.values():
            assert isinstance(provider, ProviderPlugin)

    def test_caching(self):
        """discover_providers() is cached (same dict returned)."""
        p1 = discover_providers()
        p2 = discover_providers()
        assert p1 is p2


class TestGetProvider:
    def test_get_chatgpt(self):
        """get_provider('chatgpt') returns ChatGPTProvider."""
        provider = get_provider("chatgpt")
        assert provider.id == "chatgpt"
        assert provider.name == "ChatGPT"

    def test_unknown_provider_raises(self):
        """get_provider() raises ValueError for unknown provider."""
        with pytest.raises(ValueError) as exc:
            get_provider("nonexistent")
        assert "nonexistent" in str(exc.value)
        assert "Available" in str(exc.value)

    def test_error_message_lists_available(self):
        """Error message includes list of available providers."""
        with pytest.raises(ValueError) as exc:
            get_provider("fake-provider")
        # Should list chatgpt at minimum
        assert "chatgpt" in str(exc.value).lower()

    def test_provider_is_valid_instance(self):
        """Returned provider satisfies ProviderPlugin protocol."""
        provider = get_provider("chatgpt")
        assert isinstance(provider, ProviderPlugin)
        assert hasattr(provider, "list_models")
        assert hasattr(provider, "stream")
```

| T-04-03 | Information Disclosure | config.json | accept | Config contains model preferences, not secrets. Secrets are in auth.json (handled by Phase 2 with 0o600 perms). |
