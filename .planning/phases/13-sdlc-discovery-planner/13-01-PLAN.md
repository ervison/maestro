    ```python
    """SDLC Discovery Harness — orchestrates 13-artifact specification generation."""
    from __future__ import annotations
    import asyncio
    import os
    from pathlib import Path
    from maestro.sdlc.schemas import (
        SDLCRequest, SDLCArtifact, DiscoveryResult,
        ArtifactType, ARTIFACT_ORDER, ARTIFACT_FILENAMES,
    )

    class DiscoveryHarness:
        def __init__(self, provider=None, model: str | None = None, workdir: str = "."):
            self._provider = provider
            self._model = model
            self._workdir = workdir

        def run(self, request: SDLCRequest) -> DiscoveryResult:
            return asyncio.run(self.arun(request))

        async def arun(self, request: SDLCRequest) -> DiscoveryResult:
            spec_dir = Path(request.workdir) / "spec"
            spec_dir.mkdir(parents=True, exist_ok=True)
            artifacts: list[SDLCArtifact] = []
            for artifact_type in ARTIFACT_ORDER:
                artifact = await self._generate_artifact(request, artifact_type)
                artifacts.append(artifact)
                (spec_dir / artifact.filename).write_text(artifact.content, encoding="utf-8")
            return DiscoveryResult(request=request, artifacts=artifacts, spec_dir=str(spec_dir))

        async def _generate_artifact(
            self, request: SDLCRequest, artifact_type: ArtifactType
        ) -> SDLCArtifact:
            """Generate a single artifact. Overridden by specialized generators in 13-02/13-03."""
            filename = ARTIFACT_FILENAMES[artifact_type]
            content = f"# {artifact_type.value.replace('_', ' ').title()}\n\n{request.prompt}\n"
            return SDLCArtifact(artifact_type=artifact_type, filename=filename, content=content)
    ```

|----------|-------------|
| user prompt → SDLCRequest | Untrusted string from CLI enters the system |
| SDLCRequest.workdir → filesystem | Path controls where spec/ is written |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-13-01 | Tampering | SDLCRequest.prompt | mitigate | Validate non-empty prompt in `__post_init__`; strip whitespace before check |
| T-13-02 | Elevation of Privilege | SDLCRequest.workdir → Path traversal | mitigate | In harness.arun(), resolve workdir with `Path(request.workdir).resolve()` and validate it is an absolute path inside an acceptable root before writing; do not follow symlinks outside workdir |
| T-13-03 | Denial of Service | spec_dir.mkdir(parents=True) on adversarial path | mitigate | Catch OSError/PermissionError in arun() and surface as RuntimeError with safe message |
| T-13-04 | Information Disclosure | artifact content written to spec/ | accept | spec/ is a developer-local directory; no secrets are written by the schema layer |
</threat_model>
